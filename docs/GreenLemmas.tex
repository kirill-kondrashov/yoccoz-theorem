\section{GreenLemmas}
\textit{Source: \texttt{Quadratic/Complex/GreenLemmas.lean}}

\begin{quotation}
\textbf{Convergence for escaping points}
\end{quotation}

\subsection*{escape\_bound}\label{escape_bound}
A bound used to ensure the orbit is large enough for the log approximation.

\textit{(See code: escape\_bound)}

\subsection*{potential\_seq}\label{potential_seq}
The n-th approximation of the Green's function: \texttt{1/2\textasciicircum{}n * log (max 1 ‖f\_c\textasciicircum{}n(z)‖)}.

\textit{(See code: potential\_seq)}

\subsection*{green\_function}\label{green_function}
The Green's function \texttt{G\_c(z)}. Defined as the limit of \texttt{potential\_seq}.

\textit{(See code: green\_function)}

\subsection*{potential\_seq\_converges\_of\_mem\_K}\label{potential_seq_converges_of_mem_K}
Convergence of the potential sequence to 0 for \texttt{z $\in$ K(c)}.

\textit{(See code: potential\_seq\_converges\_of\_mem\_K)}

\subsection*{potential\_seq\_converges\_of\_escapes}\label{potential_seq_converges_of_escapes}
Convergence of the potential sequence for \texttt{z $\notin$ K(c)}.

\textit{(See code: potential\_seq\_converges\_of\_escapes)}

\subsection*{potential\_seq\_converges}\label{potential_seq_converges}
Convergence of the potential sequence for all \texttt{z}.

\textit{(See code: potential\_seq\_converges)}

\subsection*{green\_function\_eq\_lim}\label{green_function_eq_lim}
\texttt{G\_c(z)} equals the limit of the potential sequence.

\textit{(See code: green\_function\_eq\_lim)}
